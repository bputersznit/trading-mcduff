### Add Stopout Lockout
```csharp
private DateTime lastStopoutTime = DateTime.MinValue;
private const int STOPOUT_LOCKOUT_SECONDS = 120;

private bool IsWithinStopoutLockout()
{
    if ((Time[0] - lastStopoutTime).TotalSeconds < STOPOUT_LOCKOUT_SECONDS)
    {
        Print($"[LOCKOUT] Signal blocked: {(int)(Time[0] - lastStopoutTime).TotalSeconds}sec since last stop");
        return true;
    }
    return false;
}

private void HandleExitExecution(Execution exec)
{
    if (exec.Order.Name == "Stop loss")
    {
        lastStopoutTime = exec.Time;  // Start lockout timer
        Print($"[LOCKOUT] Stopout detected, blocking entries for {STOPOUT_LOCKOUT_SECONDS}sec");
    }
}
```
